      1. Agent B sends request with tampered receipt (wrong signature)

      3. No balance changes on either side
    - Scenario 4 (backward compat — no receipt, same machine):

    import { describe, it, expect, beforeEach, afterEach } from 'vitest';

    describe('P2P Credit Integration (two separate SQLite databases)', () => {
      // Setup: two in-memory DBs, two keypairs, one gateway server

      let providerRegistryDb: Database.Database;
      let providerCreditDb: Database.Database;
      let requesterCreditDb: Database.Database;
      let providerKeys: { publicKey: Buffer; privateKey: Buffer };
      let requesterKeys: { publicKey: Buffer; privateKey: Buffer };
      let gateway: FastifyInstance;
      const providerOwner = 'agent-provider';
      const requesterOwner = 'agent-requester';
      const cardId = '...'; // UUID for test card

      beforeEach(() => {
        // Create separate DBs
        providerRegistryDb = openDatabase(':memory:');
        providerCreditDb = openCreditDb(':memory:');
        requesterCreditDb = openCreditDb(':memory:');

        // Bootstrap agents in their OWN DBs
        bootstrapAgent(providerCreditDb, providerOwner, 50);
        bootstrapAgent(requesterCreditDb, requesterOwner, 100);

        // Generate keypairs
        providerKeys = generateKeyPair();
        requesterKeys = generateKeyPair();

        // Insert a test card in provider's registry
        // ... (v2 card with skill pricing 5 credits)

        // Create gateway server with provider's DBs + mock SkillExecutor
        gateway = createGatewayServer({
          registryDb: providerRegistryDb,
          creditDb: providerCreditDb,
          tokens: ['test-token'],
          handlerUrl: 'http://localhost:9999',
          silent: true,
          skillExecutor: mockSkillExecutor, // Returns success
        });
      });

      it('success: requester pays, provider earns, separate DBs', async () => {
        // 1. Requester creates signed receipt on OWN DB
        const { escrowId, receipt } = createSignedEscrowReceipt(
          requesterCreditDb, requesterKeys.privateKey, requesterKeys.publicKey,
          { owner: requesterOwner, amount: 5, cardId }
        );

        // 2. Send request to provider gateway with receipt
        const response = await gateway.inject({
          method: 'POST', url: '/rpc',
          headers: { Authorization: 'Bearer test-token', 'Content-Type': 'application/json' },
          payload: { jsonrpc: '2.0', id: '1', method: 'capability.execute',
            params: { card_id: cardId, requester: requesterOwner, escrow_receipt: receipt }
          }
        });

        expect(response.statusCode).toBe(200);
        const body = response.json();
        expect(body.result).toBeDefined();

        // 3. Provider records earning (done by gateway handler)
        // Assert provider balance increased
        expect(getBalance(providerCreditDb, providerOwner)).toBe(55); // 50 + 5

        // 4. Requester settles escrow
        settleRequesterEscrow(requesterCreditDb, escrowId);
        expect(getBalance(requesterCreditDb, requesterOwner)).toBe(95); // 100 - 5
      });

      // ... failure, invalid receipt, backward compat scenarios
    });
