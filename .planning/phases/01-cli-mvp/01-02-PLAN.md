```typescript
import { Bonjour } from 'bonjour-service';
const bonjour = new Bonjour();

// Publish a service
bonjour.publish({ name: string, type: string, port: number, txt?: Record<string, string> });

// Browse for services
const browser = bonjour.find({ type: string }, callback?: (service) => void);
browser.on('up', (service) => { /* service.name, service.host, service.port, service.addresses */ });
browser.on('down', (service) => { /* service removed */ });
browser.stop();

// Cleanup
bonjour.unpublishAll(callback?: () => void);
bonjour.destroy();
```
