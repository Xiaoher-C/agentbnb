    ```typescript
    export class HubAgentExecutor {
      constructor(
        private registryDb: Database.Database,
        private creditDb: Database.Database,
      ) {}

      async execute(
        agentId: string,
        skillId: string,
        params: Record<string, unknown>,
        requesterOwner?: string,
      ): Promise<ExecutionResult>
    }
    ```

    Implementation:
    1. Load agent via `getHubAgent(registryDb, agentId, getMasterKey())`. Return error if null or status === 'paused'.
    2. Find skill route in `agent.skill_routes` matching `skillId`. Return error if not found.
    3. Switch on `route.mode`:
       - `direct_api`: Cast `route.config` to ApiSkillConfig. If the agent has encrypted secrets, decrypt them and inject into the config's `auth` field (e.g., if `secrets.api_key` exists and config.auth.type === 'bearer', set `config.auth.token = decryptedValue`). Use the `expandEnvVars`-like pattern but resolve from decrypted secrets instead of process.env. Create `new ApiExecutor()` and call `execute(config, params)`. Wrap with timing to produce `latency_ms`.
       - `relay`: Return `{ success: false, error: 'relay mode requires connected session agent', latency_ms: 0 }`.
       - `queue`: Return `{ success: false, error: 'queue mode not yet implemented (Phase 37)', latency_ms: 0 }`.
    4. If `requesterOwner` is provided, handle credit escrow: hold before execution, settle on success, release on failure. Use `holdEscrow/settleEscrow/releaseEscrow` from `src/credit/escrow.ts` with `creditDb`. The hold amount comes from the skill's `pricing.credits_per_call` in the route config.

    - Mock `fetch` (globalThis.fetch) to simulate external API responses for direct_api tests.
    - Use in-memory SQLite with `initHubAgentTable` + `createHubAgent` to set up test agents.

    - Test all mode branches and error cases.
