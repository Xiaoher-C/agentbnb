  Response: { jobs: Job[] }

GET /api/hub-agents/:id/jobs/:jobId

1. **Add Hub Agent types to `hub/src/types.ts`** (append after existing types):

```typescript
/** Hub Agent as returned by GET /api/hub-agents */
export interface HubAgentSummary {
  agent_id: string;
  name: string;
  public_key: string;
  skill_routes: HubAgentSkillRoute[];
  status: 'active' | 'paused';
  secret_keys?: string[];
  created_at: string;
  updated_at: string;
}

/** Skill routing config for a Hub Agent */
export type HubAgentSkillRoute =
  | { skill_id: string; mode: 'direct_api'; config: { id: string; type: 'api'; name: string; endpoint: string; method: string; pricing: { credits_per_call: number }; [key: string]: unknown } }
  | { skill_id: string; mode: 'relay'; config: { relay_owner: string } }
  | { skill_id: string; mode: 'queue'; config: { relay_owner: string } };

/** Job record from GET /api/hub-agents/:id/jobs */
export interface HubAgentJob {
  id: string;
  hub_agent_id: string;
  skill_id: string;
  requester_owner: string;
  params: string;
  status: 'queued' | 'dispatched' | 'completed' | 'failed';
  result: string | null;
  escrow_id: string | null;
  relay_owner: string | null;
  created_at: string;
  updated_at: string;
}
```

2. **Create `hub/src/hooks/useHubAgents.ts`** with three hooks following the established isFirstFetch polling pattern (30s interval, no flicker):

- `useHubAgents()`: Fetches `GET /api/hub-agents`, returns `{ agents: HubAgentSummary[], loading, error }`. Polls every 30s.
- `useHubAgent(agentId: string)`: Fetches `GET /api/hub-agents/:id`, returns `{ agent: HubAgentSummary | null, loading, error }`. Polls every 30s.
- `useHubAgentJobs(agentId: string)`: Fetches `GET /api/hub-agents/:id/jobs`, returns `{ jobs: HubAgentJob[], loading, error }`. Polls every 10s (jobs change faster).

Follow the exact isFirstFetch + useRef pattern from `hub/src/hooks/useAgents.ts`. Use the same graceful degradation (keep existing data on error).
