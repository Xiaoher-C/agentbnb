      { name: 'agents', description: 'Agent profiles and reputation' },
      { name: 'identity', description: 'Agent identity and guarantor registration' },
      { name: 'owner', description: 'Owner-only endpoints (Bearer auth required)' },
      { name: 'system', description: 'Health and stats' },
      { name: 'pricing', description: 'Market pricing statistics' },
    ],
    components: {
      securitySchemes: {
        bearerAuth: { type: 'http', scheme: 'bearer' },
        ed25519Auth: {
          type: 'apiKey',
          in: 'header',
          name: 'X-Agent-PublicKey',
          description: 'Ed25519 public key (hex). Also requires X-Agent-Signature and X-Agent-Timestamp headers.',
        },
      },
    },
  },

await server.register(swaggerUi, {

Fastify queues plugin registrations and resolves them before `server.listen()`. Do NOT change the function signature to async.

**Step 1: Create src/registry/openapi-gpt-actions.ts**

This module converts the auto-generated OpenAPI 3.0 spec into GPT Actions format. GPT Actions format is a subset of OpenAPI 3.0 with specific requirements:

```typescript
/**
 * Converts a standard OpenAPI 3.0 spec into GPT Actions-compatible format.
 *
 * GPT Actions differences from standard OpenAPI:
 * 1. Only GET and POST methods supported (strip DELETE, PATCH, PUT)
 * 2. `servers` must have exactly one entry with absolute URL
 * 3. Remove security schemes GPT can't handle (Ed25519 custom headers)
 * 4. Strip internal/owner-only endpoints (paths starting with /me, /draft)
 * 5. Add `operationId` to every operation if missing
 * 6. Flatten $ref references (GPT Builder prefers inline schemas)
 */
export function convertToGptActions(
  openapiSpec: Record<string, unknown>,
  serverUrl: string,
): Record<string, unknown>
```

Implementation:
- Deep-clone the input spec (JSON parse/stringify)
- Set `servers` to `[{ url: serverUrl }]`
- Filter `paths` to only include GET and POST methods
- Remove paths that start with `/me` or `/draft` or `/docs` or `/ws` (internal/owner-only)
- Remove paths under `/api/credits/` (require Ed25519 — GPT can't do custom header signing)
- For remaining paths, ensure each operation has an `operationId` (derive from method + path, e.g., `getCards`, `postCards`, `getApiPricing`)
- Remove `security` from individual operations (GPT uses its own auth mechanism)
- Remove `securitySchemes` from components
- Keep `info`, `openapi` version, `tags` (filter to relevant tags only)

**Step 2: Add GET /api/openapi/gpt-actions route in server.ts**

```typescript
import { convertToGptActions } from './openapi-gpt-actions.js';
```

Add route (public, no auth):
```typescript
server.get('/api/openapi/gpt-actions', {
  schema: {
    tags: ['system'],
    summary: 'GPT Actions-compatible OpenAPI schema',
    description: 'Returns a GPT Builder-importable OpenAPI spec with only public GET/POST endpoints',
    querystring: {
      type: 'object',
      properties: {
        server_url: { type: 'string', description: 'Base URL for the server (required for absolute URLs in GPT Actions)' },
      },
    },
    response: { 200: { type: 'object', additionalProperties: true } },
  },
}, async (request, reply) => {
  const query = request.query as Record<string, string | undefined>;
  const serverUrl = query.server_url?.trim() || `${request.protocol}://${request.hostname}`;
  const openapiSpec = server.swagger();
  const gptActions = convertToGptActions(openapiSpec as Record<string, unknown>, serverUrl);
  return reply.send(gptActions);
});
```

Note: `server.swagger()` returns the generated OpenAPI spec object. It is available after `await server.ready()` in production, but in route handlers Fastify is already ready.
